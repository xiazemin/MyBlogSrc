三、getTraceAsString()
Exception类中的getTraceAsString()成员函数返回一个调用堆栈

//调用函数parent_func
function child_func() {
    parent_func();
}
 
//调用grandparent_func
function parent_func() {
    grandparent_func();
}
 
//打印调用堆栈
function grandparent_func() {
    $e = new Exception; 
    var_dump($e->getTraceAsString()); 
}
 
//主函数调用
child_func();
